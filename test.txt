int getNum1(int num) {
	int cnt = 0;

	while (num != 0) {
		int tmp = num;
		num = num>1;
		if (tmp&num){
			cnt++;
		}
	}
	return cnt;
}


boolean compareTwoBinaryTree(BinaryTree tree1, BinaryTree tree2) {

	if (tree1.val != tree2.val) 
		return false;
		
	while (tree1.left != null) {
	BinaryTree tmp1 =tree1.left;
	BinaryTree tmp2 =tree2.left;
		if (!tmp1.val == tmp2.val) 
		return false;
	}
	
	while (tree1.right != null) {
	BinaryTree tmp1 =tree1.right;
	BinaryTree tmp2 =tree2.right;
		if (!tmp1.val == tmp2.val) 
		return false;
	}
	
	return true;
}
